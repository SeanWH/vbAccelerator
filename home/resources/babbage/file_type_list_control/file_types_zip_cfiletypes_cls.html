VERSION 1.0 CLASS
BEGIN
  MultiUse = -1  'True
END
Attribute VB_Name = "CFileTypes"
Attribute VB_GlobalNameSpace = False
Attribute VB_Creatable = False
Attribute VB_PredeclaredId = False
Attribute VB_Exposed = True
Option Explicit

Private m_hWnd As Long

Friend Sub fInit(ByVal hWnd As Long)
   m_hWnd = hWnd
End Sub

Public Property Get Count() As Long
Dim ctl As vbalFileType
   If gbVerify(m_hWnd, ctl) Then
      Count = ctl.fCount
   End If
End Property
Public Property Get Item(Index As Variant) As CFileType
Dim ctl As vbalFileType
Dim lindex As Long
   If gbVerify(m_hWnd, ctl) Then
      lindex = ctl.fGetIndex(Index)
      If lindex > 0 Then
         Dim cF As New CFileType
         cF.fInit m_hWnd, lindex
         Set Item = cF
      End If
   End If
End Property
Public Function Add( _
      ByVal sExt As String, _
      ByVal sProgId As String _
   ) As CFileType
Dim ctl As vbalFileType
Dim lindex As Long
   If gbVerify(m_hWnd, ctl) Then
      lindex = ctl.fAdd(sExt, sProgId)
      If lindex > 0 Then
         Dim cF As New CFileType
         cF.fInit m_hWnd, lindex
         Set Add = cF
      End If
   End If
End Function

Public Sub Remove(Index As Variant)
Dim ctl As vbalFileType
Dim lindex As Long
   If gbVerify(m_hWnd, ctl) Then
      lindex = ctl.fGetIndex(Index)
      If lindex > 0 Then
         ctl.fRemoveFileType lindex
      End If
   End If
End Sub
