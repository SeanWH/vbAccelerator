Private Sub cmdGetPic_Click()
   ' Transfer an image to a VB picture object. StdPicture sucks
   ' for icons!
   Set picIcon.Picture = m_cIL.ItemPicture(m_sFirstFile)
   
End Sub

   ' This will crash VB unless the icon is 16x16 or 32x32.  Why?
   SavePicture picIcon, App.Path & "\Test.ico"
End Sub

Private Sub cmdShow_Click()
Dim i As Long
Dim x As Long, y As Long
Dim bSel As Boolean, bDis As Boolean, bCut As Boolean

Dim hFiles As Long
Dim fd As WIN32_FIND_DATA
Dim sPath As String, sName As String
Dim f As Boolean

   Me.Cls
   m_sFirstFile = ""
   x = cmdShow.left + cmdShow.Width
   x = x \ Screen.TwipsPerPixelX
   y = cmdShow.tOp \ Screen.TwipsPerPixelY
   bSel = optStyle(1).Value
   bDis = optStyle(2).Value
   bCut = optStyle(3).Value
   'sPath = "C:\WINDOWS\SYSTEM\" ' (It does work!)
   'sPath = "C:\Stevemac\Icon Library\Toons\Ren & Stimpy\"
   sPath = App.Path & "\"
   lblPath.Caption = sPath

   ' Display the icons for the file in this directory:
   hFiles = FindFirstFile(sPath & "*.*", fd)
   f = Not (hFiles = INVALID_HANDLE_VALUE)
   Do While f
      sName = ByteZToStr(fd.cFileName)
      ' Skip . and ..
      If left$(sName, 1) <> "." Then
         If m_sFirstFile = "" Then
            m_sFirstFile = sPath & sName
         End If
         ' draw it:
         m_cIL.DrawImage sPath & sName, Me.hdc, x, y, bSel, bCut, bDis,
          Me.BackColor
         
         ' Next Position:
         y = y + m_cIL.IconSizeY + 2
         If (y + m_cIL.IconSizeY + 2 > cmdStrip.tOp \ Screen.TwipsPerPixelY)
          Then
            y = cmdShow.tOp \ Screen.TwipsPerPixelY
            x = x + m_cIL.IconSizeX + 2
         End If
      End If
      
      ' Keep looping until no more files
       f = FindNextFile(hFiles, fd)
   Loop
   f = FindClose(hFiles)
      
End Sub
Private Function ByteZToStr(ByRef b() As Byte) As String
Dim iPos As Long
Dim s As String
   s = StrConv(b, vbUnicode)
   iPos = InStr(s, vbNullChar)
   If iPos > 0 Then
      ByteZToStr = left$(s, iPos - 1)
   Else
      ByteZToStr = s
   End If
End Function

Private Sub cmdStrip_Click()
Dim sPath As String, sName As String
Dim hFiles As Long, f As Boolean
Dim v() As Variant, iC As Long
Dim fd As WIN32_FIND_DATA

   sPath = App.Path & "\"

   ' Display the icons for the file in this directory:
   hFiles = FindFirstFile(sPath & "*.*", fd)
   f = Not (hFiles = INVALID_HANDLE_VALUE)
   Do While f
      sName = ByteZToStr(fd.cFileName)
      ' Skip . and ..
      If left$(sName, 1) <> "." Then
         iC = iC + 1
         ReDim Preserve v(1 To iC) As Variant
         v(iC) = sPath & sName
      End If
      ' Keep looping until no more files
       f = FindNextFile(hFiles, fd)
   Loop
   f = FindClose(hFiles)
   
   Set picStrip.Picture = m_cIL.ImagePictureStrip(v(), &H80FF00)
End Sub

   lblPath.Caption = App.Path
