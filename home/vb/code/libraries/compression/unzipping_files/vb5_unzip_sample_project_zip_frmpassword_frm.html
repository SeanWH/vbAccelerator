Private Sub cmdOK_Click()
Dim sPwd As String
   sPwd = Trim$(txtPassword.Text)
   If Len(sPwd) > 0 Then
      m_sPassword = sPwd
      m_bCancel = False
      Unload Me
   Else
      MsgBox "Please enter a password.", vbInformation
      txtPassword.SetFocus
   End If
End Sub
