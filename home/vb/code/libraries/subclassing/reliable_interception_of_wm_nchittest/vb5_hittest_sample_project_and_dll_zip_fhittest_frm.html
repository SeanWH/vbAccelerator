Attribute VB_Name = "frmHitTest"
Attribute VB_GlobalNameSpace = False
Attribute VB_Creatable = False
Attribute VB_PredeclaredId = True
Attribute VB_Exposed = False
Option Explicit

Private m_cHitTest As New cHitTester

Private Sub Form_Load()
   With m_cHitTest
      ' Start hit testing:
      .AttachHitTester Me
      
      ' Add an area for the pretend "title bar"
      .AddArea Me.hWnd, "TitleBar"
      ' Set the title bar's area:
      .SetTestAreaFromObject "TitleBar", lblTest, HTCAPTION
      
      ' Add another title bar area:
      .AddArea Me.hWnd, "vbAccelerator"
      ' Set the title bar's area:
      .SetTestAreaFromObject "vbAccelerator", imgVbAccel, HTCAPTION
      
      ' Add an area for the resize box:
      .AddArea Me.hWnd, "SizeBox"
      ' Set the area:
      .SetTestAreaFromObject "SizeBox", lblSizeBox(0), HTBOTTOMRIGHT
   End With
   
End Sub

Private Sub Form_QueryUnload(Cancel As Integer, UnloadMode As Integer)
   ' Stop hit testing:
   m_cHitTest.DestroyHitTester
   Set m_cHitTest = Nothing
End Sub

Private Sub Form_Resize()
   ' Move the size box:
   lblSizeBox(0).Move Me.ScaleWidth - lblSizeBox(0).Width, Me.ScaleHeight -
    lblSizeBox(0).Height
   lblSizeBox(1).Move lblSizeBox(0).Left - 15, lblSizeBox(0).Top
   ' Resize the hit test rectangle in the DLL:
   m_cHitTest.SetTestAreaFromObject "SizeBox", lblSizeBox(0), HTBOTTOMRIGHT
End Sub
