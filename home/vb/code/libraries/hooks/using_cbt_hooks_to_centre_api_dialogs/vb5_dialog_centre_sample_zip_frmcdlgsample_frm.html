Private Function ReadFileText(ByVal sFile As String) As String
On Error GoTo ErrorHandler
Dim iFile As Integer
Dim sBuf As String
   iFile = FreeFile
   Open sFile For Binary Access Read Lock Write As #iFile
   sBuf = Space$(LOF(iFile))
   Get #iFile, , sBuf
   Close #iFile
   ReadFileText = sBuf
   Exit Function
ErrorHandler:
   MsgBox "An error occurred trying to read '" & sFile & "'" & vbCrLf & "[" &
    Err.Number & "] :" & Err.Description
   Close #iFile
   Exit Function
End Function
Private Sub WriteFileText(ByVal sFile As String, ByVal sText As String)
On Error GoTo ErrorHandler
Dim iFile As Integer
   iFile = FreeFile
   Open sFile For Binary Access Write Lock Read As #iFile
   Put #iFile, , sText
   Close #iFile
   Exit Sub
ErrorHandler:
   MsgBox "An error occurred trying to read '" & sFile & "'" & vbCrLf & "[" &
    Err.Number & "] :" & Err.Description
   Close #iFile
   Exit Sub
End Sub


Private Sub Form_Resize()
   On Error Resume Next
   fraSep.Move -6 * Screen.TwipsPerPixelX, -3 * Screen.TwipsPerPixelY,
    Me.ScaleWidth + 12 * Screen.TwipsPerPixelX
   txtDoc.Move txtDoc.Left, txtDoc.Top, Me.ScaleWidth - txtDoc.Left * 2,
    Me.ScaleHeight - txtDoc.Top - 2 * Screen.TwipsPerPixelY
End Sub

Private Sub mnuFile_Click(Index As Integer)
Dim cD As cCommonDialog
Dim cC As cCommonDialogCentre
Dim sFile As String
   Select Case Index
   Case 0
      Set cC = New cCommonDialogCentre
      If (chkCentre.Value = Checked) Then
         cC.Start Me, 640, 480
      End If
      Set cD = New cCommonDialog
      If cD.VBGetOpenFileName( _
            sFile, _
            Filter:="Text Files (*.txt)|*.txt|All Files (*.*)|*.*", _
            DefaultExt:="TXT", _
            Owner:=Me.hwnd, _
            flags:=OFN_HIDEREADONLY) Then
         txtDoc.Text = ReadFileText(sFile)
      End If
   Case 1
      Set cC = New cCommonDialogCentre
      If (chkCentre.Value = Checked) Then
         cC.Start Me, 640, 480
      End If
      Set cD = New cCommonDialog
      If cD.VBGetSaveFileName(sFile, _
            Filter:="Text Files (*.txt)|*.txt|All Files (*.*)|*.*", _
            DefaultExt:="TXT", _
            Owner:=Me.hwnd) Then
         WriteFileText sFile, txtDoc.Text
      End If
   Case 3
      Set cC = New cCommonDialogCentre

         cC.Start Me
      End If

      End If

      Set cC = New cCommonDialogCentre

      End If
