'Declarations for AsyncStart:

Private mcolRunnables As Collection
Private WithEvents moTimer As objTimer
Attribute moTimer.VB_VarHelpID = -1
Private Declare Function CoLockObjectExternal Lib "ole32" ( _
    ByVal pUnk As IUnknown, ByVal fLock As Long, _
    ByVal fLastUnlockReleases As Long) As Long

'-------------------------------------------------------------------------------
'Declarations for Sleep:

Private Declare Sub SleepAPI Lib "kernel32" Alias "Sleep" ( _
    ByVal dwMilliseconds As Long)

'-------------------------------------------------------------------------------
'Initialization and cleanup:

Private Sub Class_Initialize()
    Set mcolRunnables = New Collection
    Set moTimer = New objTimer
    moTimer.Interval = 1
End Sub

'-------------------------------------------------------------------------------
'AsyncStart:

Public Sub Start(Runnable As Runnable)
    'Ask the system to lock the object so that it will still perform its work
     even if it is released
    CoLockObjectExternal Runnable, 1, 1
    mcolRunnables.Add Runnable
    moTimer.Enabled = True
End Sub

Private Sub moTimer_Timer()
    Dim oRunnable As Runnable

    moTimer.Enabled = False
    With mcolRunnables
        Do While .Count > 0
            Set oRunnable = .Item(1)
'-------------------------------------------------------------------------------
            'I must remove the item before I call oRunnable.Start. This is
            '    because oRunnable.Start may in one way or another cause another
            '    Runnable to call Start Me and ultimately end up here. If I
            '    didn't remove this item in advance, the other Runnable would
            '    then try to act on it, calling its Start and removing it.
            '    Not only would Start be called twice for the same runnable, but
            '    eventually I would try to remove the same item twice, resulting
            '    in an error the second time. Guess how I know all this? :)
            .Remove 1
            oRunnable.Start
            'Ask the system to release its lock on the object
            CoLockObjectExternal oRunnable, 0, 1
        Loop
    End With
End Sub

'-------------------------------------------------------------------------------
'Shelling:

Public Function SyncShell(CommandLine As String, Optional Timeout As Long, _
    Optional WaitForInputIdle As Boolean) As Boolean

    Dim proc As PROCESS_INFORMATION
    Dim Start As STARTUPINFO
    Dim ret As Long
    Dim nMilliseconds As Long

    If Timeout > 0 Then
        nMilliseconds = Timeout
    Else
        nMilliseconds = INFINITE
    End If

    'Initialize the STARTUPINFO structure:
    Start.cb = Len(Start)

    'Start the shelled application:
    CreateProcessA 0&, CommandLine, 0&, 0&, 1&, _
        NORMAL_PRIORITY_CLASS, 0&, 0&, Start, proc

    If WaitForInputIdle Then
        'Wait for the shelled application to finish setting up its UI:
        ret = InputIdle(proc.hProcess, nMilliseconds)
    Else
        'Wait for the shelled application to terminate:
        ret = WaitForSingleObject(proc.hProcess, nMilliseconds)
    End If

    CloseHandle proc.hProcess

    'Return True if the application finished. Otherwise it timed out or erred.
    SyncShell = (ret = WAIT_OBJECT_0)
End Function

'-------------------------------------------------------------------------------
'Sleep:

Public Sub Sleep(Milliseconds As Long)
    SleepAPI Milliseconds
End Sub
